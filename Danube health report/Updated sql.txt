CREATE TABLESPACE ORION11JTBS1
  DATAFILE 'USERS1.DBF' 
    SIZE 10M
    REUSE
    AUTOEXTEND ON NEXT 10M MAXSIZE 200M;


CREATE USER REPDR IDENTIFIED BY REPDR DEFAULT TABLESPACE ORION11JTBS1 TEMPORARY TABLESPACE TEMP QUOTA UNLIMITED ON ORION11JTBS1; 

GRANT CONNECT,DBA,CREATE SESSION TO REPDR;

CREATE TABLE DR_REP_DBA
(
DR_REP_QRY VARCHAR2(1000),
DR_DATE DATE DEFAULT SYSDATE
);



CREATE TABLE DB_CUSTOMER_DETAILS(
        DB_CUSTNAME   VARCHAR2(50),
        DB_LOCATION   VARCHAR2(15),
        DB_NAME       VARCHAR2(10),
        DB_ID         VARCHAR2(20),
        DB_VERSION    VARCHAR2(50));

CREATE TABLE DB_HD_DETAILS(        
        DB_HOSTNAME VARCHAR2(20),
        DB_IP       VARCHAR2(10),
        DB_OS       VARCHAR2(20),
        DB_MAKE     VARCHAR2(50),
        DB_MODEL    VARCHAR2(50),
        DB_CPU_TYPE VARCHAR2(20),
        DB_CORE_COUNT VARCHAR2(20),
        DB_RAM      VARCHAR2(20));

CREATE TABLE DB_DR_PARAMETERS(
        DB_PARAMETER  VARCHAR2(20),
        DB_DESCRIPTION VARCHAR2(50));


CREATE TABLE DB_ARC_LOCATION(
        DB_DC_ARC_LOC VARCHAR2(30),
        DB_DR_ARC_LOC VARCHAR2(30),
        DB_DR_TRC_LOC VARCHAR2(30));

INSERT INTO DR_REP_DBA VALUES('SELECT DB_CUSTNAME as"Customer Name",DB_LOCATION as"Geos/Location" FROM DB_CUSTOMER_DETAILS WHERE Trunc(DB_DATE)=Trunc(SYSDATE)');

INSERT INTO DR_REP_DBA VALUES('SELECT DB_HOSTNAME  as"Host Name",DB_IP as"IP",DB_OS as"OS",DB_MAKE as"Make",DB_MODEL as"Model",DB_CPU_TYPE as"CPU type",DB_CORE_COUNT as"CPU core count", DB_RAM as"RAM(in GB)" FROM DB_HD_DETAILS WHERE Trunc(DB_DATE)=Trunc(SYSDATE)');

INSERT INTO DR_REP_DBA VALUES('SELECT DB_NAME as"DB Name",DB_ID as"DB ID",DB_VERSION as"Database Version / Edition" FROM DB_CUSTOMER_DETAILS WHERE Trunc(DB_DATE)=Trunc(SYSDATE)');

INSERT INTO DR_REP_DBA VALUES('SELECT DB_PARAMETER  as"Parameters",DB_DESCRIPTION as"Description" FROM DB_DR_PARAMETERS WHERE Trunc(DB_DATE)=Trunc(SYSDATE)');

INSERT INTO DR_REP_DBA VALUES('SELECT DB_DC_ARC_LOC  as"DC archive log destination",DB_DR_ARC_LOC as"DR archive log destination" FROM DB_ARC_LOCATION WHERE Trunc(DB_DATE)=Trunc(SYSDATE)');

INSERT INTO DR_REP_DBA VALUES('SELECT DB_DR_TRC_LOC  as"Folder name" FROM DB_ARC_LOCATION WHERE Trunc(DB_DATE)=Trunc(SYSDATE)');
